CREATE
VIEW VIEW_PESSOA_TRANSPORTADORA
AS

select 
	T.ID,
	P.NOME, P.TIPO, P.EMAIL, P.SITE, 
	PF.CPF AS "CPF_CNPJ", 
	PF.RG AS "RG_IE",
	T.DATA_CADASTRO,
	T.OBSERVACAO,
	P.ID as "ID_PESSOA"
from
	PESSOA P 
	INNER JOIN PESSOA_FISICA PF ON (PF.ID_PESSOA = P.ID)
	INNER JOIN TRANSPORTADORA T ON (T.ID_PESSOA = P.ID)
where 
	P.TRANSPORTADORA = "S"

UNION

select 
	T.ID,
	P.NOME, P.TIPO, P.EMAIL, P.SITE, 
	PJ.CNPJ AS "CPF_CNPJ", 
	PJ.INSCRICAO_ESTADUAL AS "RG_IE",
	T.DATA_CADASTRO,
	T.OBSERVACAO,
	P.ID as "ID_PESSOA"
from
	PESSOA P 
	INNER JOIN PESSOA_JURIDICA PJ ON (PJ.ID_PESSOA = P.ID)
	INNER JOIN TRANSPORTADORA T ON (T.ID_PESSOA = P.ID)
where 
	P.TRANSPORTADORA = "S"
