CREATE
VIEW VIEW_PESSOA_CLIENTE
AS

select 
	C.ID,
	P.NOME, P.TIPO, P.EMAIL, P.SITE, 
	PF.CPF AS "CPF_CNPJ", 
	PF.RG AS "RG_IE",
	C.DESDE,
	C.TAXA_DESCONTO,
	C.LIMITE_CREDITO,
	C.DATA_CADASTRO,
	C.OBSERVACAO,
	P.ID as "ID_PESSOA"
from
	PESSOA P 
	INNER JOIN PESSOA_FISICA PF ON (PF.ID_PESSOA = P.ID)
	INNER JOIN CLIENTE C ON (C.ID_PESSOA = P.ID)
where 
	P.CLIENTE = "S"

UNION

select 
	C.ID,
	P.NOME, P.TIPO, P.EMAIL, P.SITE, 
	PJ.CNPJ AS "CPF_CNPJ", 
	PJ.INSCRICAO_ESTADUAL AS "RG_IE",
	C.DESDE,
	C.TAXA_DESCONTO,
	C.LIMITE_CREDITO,
	C.DATA_CADASTRO,
	C.OBSERVACAO,
	P.ID as "ID_PESSOA"
from
	PESSOA P 
	INNER JOIN PESSOA_JURIDICA PJ ON (PJ.ID_PESSOA = P.ID)
	INNER JOIN CLIENTE C ON (C.ID_PESSOA = P.ID)
where 
	P.CLIENTE = "S"
